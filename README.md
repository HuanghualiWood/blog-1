```YAML
# _conflg.yml
deploy:
  type: git
  repo:
    github: git@github.com:wongqingbin/wongqingbin.github.io.git
    gitee: git@gitee.com:wongqingbin/wongqingbin.git
  branch: master
```

```shell
# hexo部署 hexo clean && hexo deploy
hexo clean
hexo generate
hexo deploy
# hexo g -d
```
